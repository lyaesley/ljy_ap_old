CREATE TABLE SOCCER_MATCH
(
  SEQ NUMBER NOT NULL
, DIV VARCHAR2(100) NOT NULL
, MATCH_DY DATE NOT NULL
, HOME  VARCHAR2(100) NOT NULL
, AWAY VARCHAR2(100) NOT NULL
, FTR VARCHAR2(100) NOT NULL
, FTHG NUMBER(3) DEFAULT 0 NOT NULL
, FTAG NUMBER(3) DEFAULT 0 NOT NULL
, HTHG NUMBER(3) DEFAULT 0 NOT NULL
, HTAG NUMBER(3) DEFAULT 0  NOT NULL
, HTR VARCHAR2(100) NOT NULL
);

CREATE SEQUENCE SOCCER_MATCH_SEQ MINVALUE 1 MAXVALUE 999999999999999999 INCREMENT BY 1 NOORDER NOCYCLE;

CREATE INDEX SOCCER_MATCH_IDX_MATCH_DY ON SOCCER_MATCH(DIV, MATCH_DY DESC);
CREATE INDEX SOCCER_MATCH_IDX_HOME ON SOCCER_MATCH(DIV, HOME DESC);
CREATE INDEX SOCCER_MATCH_IDX_AWAY ON SOCCER_MATCH(DIV, AWAY DESC);

COMMENT ON TABLE SOCCER_MATCH IS '축구 매치 기록';
COMMENT ON COLUMN SOCCER_MATCH.SEQ IS '고유번호';
COMMENT ON COLUMN SOCCER_MATCH.DIV IS '리그이름'; 
COMMENT ON COLUMN SOCCER_MATCH.MATCH_DY IS '매치날짜';
COMMENT ON COLUMN SOCCER_MATCH.HOME IS '홈팀';
COMMENT ON COLUMN SOCCER_MATCH.AWAY IS '어웨이팀';
COMMENT ON COLUMN SOCCER_MATCH.FTR IS '풀타임 결과';
COMMENT ON COLUMN SOCCER_MATCH.FTHG IS '풀타임 홈팀 골';
COMMENT ON COLUMN SOCCER_MATCH.FTAG IS '풀타임 어웨이팀 골';
COMMENT ON COLUMN SOCCER_MATCH.HTHG IS '하프타임 홈팀 골';
COMMENT ON COLUMN SOCCER_MATCH.HTAG IS '하프타임 어웨이팀 골';
COMMENT ON COLUMN SOCCER_MATCH.HTR IS '하프타임 결과';
